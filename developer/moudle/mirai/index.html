    class Permission:
    """
    插件的权限类,
    一个插件和一个bot对应该类一个对象,
    默认都为全开放模式,
    BLACK_LIST_MODE 黑名单模式,
    WHITE_LIST_MODE 白名单模式,
    ALL_MODE 全开放模式,
    """
    BLACK_LIST_MODE = 0
    WHITE_LIST_MODE = 1
    ALL_MODE = 2

    @property
    def group_list(self) -> set:
        """
        :return: 群权限列表
        """
        pass

    @property
    def friend_list(self) -> set:
        """
        :return: 好友权限列表
        """
        pass

    @property
    def admin_list(self) -> set:
        """
        :return: 管理员权限列表
        """
        pass

    def set_group_mode(self, mode: int):
